```python
def solution(answers):
    answer = []
    a = [1,2,3,4,5]
    b = [2,1,2,3,2,4,2,5]
    c = [3,3,1,1,2,2,4,4,5,5]
    result = [0,0,0]
    
    for i in range(len(answers)):
        if a[i%5] == answers[i]:
            result[0] += 1
        if b[i%8] == answers[i]:
            result[1] += 1
        if c[i%10] == answers[i]:
            result[2] += 1
            
    for idx, r in enumerate(result):
        if r == max(result):
            answer.append(idx+1)
            
    
    return answer
```
