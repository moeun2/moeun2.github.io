```python
def solution(clothes):
    answer = 1
    hash = {}
    for i in clothes:
        hash[i[1]] = hash.get(i[1], 0) + 1
    hashList = []
    for i in hash:
        hashList.append(hash.get(i)+1)
    for i in hashList:
        answer *= i
    return answer-1
```
