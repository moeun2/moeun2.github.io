```python
def solution(n, lost, reserve):
    
    # 교집합 제거
    _reserve = list(set(reserve) - set(lost))
    _lost = list(set(lost) - set(reserve))
    
    # 정렬
    _lost.sort()
    _reserve.sort()
    
    # 빌려줄 수 있다면 제거
    for i in _reserve:
        front = i-1
        back = i+1
        
        if(front in _lost):
            _lost.remove(front)
        elif(back in _lost):
            _lost.remove(back)
    
    # 전체 학생 수 - 체육복 못 빌린 학생 수
    return n - len(_lost)
```
